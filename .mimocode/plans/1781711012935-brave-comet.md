# Part A: Babylon.js 引擎迁移实现计划

## 概述

将游戏引擎从 Three.js 迁移到 Babylon.js，同时：
- 使用 Kenney Mini Characters GLB 模型（character-male-a.glb 作为玩家，character-male-c.glb 作为敌人）
- 保持所有游戏数值完全一致
- 与现有的 Part B HUD 组件集成
- 维持 TypeScript 严格模式兼容性

## 角色模型选择

基于 Kenney Mini Characters 资源（`src/assets/kenney_mini-characters/Models/GLB format/`）：

| 用途 | 模型文件 | 顶点数 | 说明 |
|------|---------|--------|------|
| **玩家** | character-male-a.glb | 1259 | 默认/中性办公室职员外观 |
| **敌人/Boss** | character-male-c.glb | 1455 | 不同外观，形成对比 |

所有模型：
- 包含 32 个动画（idle, walk, kick 等）
- 共享纹理：colormap.png
- CC0 许可证（公共领域）

### 创建文件：

#### `src/babylon/core/EngineContext.ts`
替换 `SceneManager.ts`。单例模式，持有：
- `BABYLON.Engine`（包装 canvas）
- `BABYLON.Scene`（替换 `THREE.Scene`）
- `BABYLON.FreeCamera`（替换 `THREE.PerspectiveCamera`）
- 光照设置（环境光 + 方向光 + 点光源）
- 调整大小处理
- 渲染循环（`engine.runRenderLoop`）

关键迁移点：
- Three.js `PCFSoftShadowMap` → Babylon.js `ShadowGenerator` + `usePercentageCloserFiltering = true`
- Three.js `ACESFilmicToneMapping` → Babylon.js `imageProcessingConfiguration.toneMapping = ACES`
- Three.js `Fog` → Babylon.js `scene.fogMode = FOGMODE_EXP2`

```ts
export class EngineContext {
  private engine: BABYLON.Engine
  private scene: BABYLON.Scene
  private camera: BABYLON.FreeCamera
  private shadowGenerator: BABYLON.ShadowGenerator

  constructor(container: HTMLElement) {
    const canvas = document.createElement('canvas')
    container.appendChild(canvas)
    
    this.engine = new BABYLON.Engine(canvas, true, {
      preserveDrawingBuffer: true,
      stencil: true
    })
    this.scene = new BABYLON.Scene(this.engine)
    
    this.setupCamera()
    this.setupLighting()
    this.setupFog()
  }

  private setupCamera(): void {
    // 俯视跟随相机，匹配当前：position(0,10,15), lookAt(0,0,0)
    this.camera = new BABYLON.FreeCamera('camera', new BABYLON.Vector3(0, 10, 15), this.scene)
    this.camera.setTarget(BABYLON.Vector3.Zero())
  }

  private setupLighting(): void {
    // 环境光：color 0xfffaed, intensity 0.6
    const hemispheric = new BABYLON.HemisphericLight('ambient', new BABYLON.Vector3(0, 1, 0), this.scene)
    hemispheric.diffuse = new BABYLON.Color3(1, 0.98, 0.93)
    hemispheric.intensity = 0.6

    // 方向光：position(10,20,10), intensity 1.0, 阴影
    const dirLight = new BABYLON.DirectionalLight('dir', new BABYLON.Vector3(-0.5, -1, -0.5), this.scene)
    dirLight.intensity = 1.0
    this.shadowGenerator = new BABYLON.ShadowGenerator(2048, dirLight)
    this.shadowGenerator.usePercentageCloserFiltering = true
    this.shadowGenerator.bias = -0.0005

    // 点光源：position(0,10,0), intensity 0.3
    const pointLight = new BABYLON.PointLight('point', new BABYLON.Vector3(0, 10, 0), this.scene)
    pointLight.intensity = 0.3
  }

  getScene(): BABYLON.Scene { return this.scene }
  getEngine(): BABYLON.Engine { return this.engine }
  getCamera(): BABYLON.FreeCamera { return this.camera }
  getShadowGenerator(): BABYLON.ShadowGenerator { return this.shadowGenerator }

  update(): void { this.scene.render() }
  resize(): void { this.engine.resize() }
  dispose(): void { this.engine.dispose() }
}
```

```ts
export class AssetManager {
  private scene: BABYLON.Scene
  private meshes = new Map<string, BABYLON.Mesh>()
  private skeletons = new Map<string, BABYLON.Skeleton>()
  private animationGroups = new Map<string, BABYLON.AnimationGroup[]>()
  private shadowGenerator: BABYLON.ShadowGenerator

  constructor(scene: BABYLON.Scene, shadowGenerator: BABYLON.ShadowGenerator) {
    this.scene = scene
    this.shadowGenerator = shadowGenerator
  }

  async loadCharacter(key: string, glbPath: string): Promise<{
    root: BABYLON.Mesh
    skeleton: BABYLON.Skeleton
    animationGroups: BABYLON.AnimationGroup[]
  }> {
    const result = await BABYLON.SceneLoader.ImportMeshAsync(
      '', '', glbPath, this.scene
    )
    
    const root = result.meshes[0] as BABYLON.Mesh
    const skeleton = result.skeletons[0]
    
    // 启用阴影
    result.meshes.forEach(mesh => {
      mesh.receiveShadows = true
      this.shadowGenerator.addShadowCaster(mesh)
    })

    this.meshes.set(key, root)
    if (skeleton) this.skeletons.set(key, skeleton)
    if (result.animationGroups) this.animationGroups.set(key, result.animationGroups)

    return { root, skeleton, animationGroups: result.animationGroups }
  }

  getModel(key: string): BABYLON.Mesh | null {
    return this.meshes.get(key)?.clone(key + '_clone') ?? null
  }

  dispose(): void {
    this.meshes.forEach(m => m.dispose())
    this.meshes.clear()
  }
}
```

#### `src/babylon/core/StateMachine.ts`
敌人状态机（从 Enemy.ts 提取内联状态逻辑）。

```ts
export interface State {
  enter(): void
  update(delta: number): void
  exit(): void
}

export class StateMachine {
  private states = new Map<string, State>()
  private currentState: State | null = null
  private currentStateName: string = ''

  addState(name: string, state: State): void {
    this.states.set(name, state)
  }

  setState(name: string): void {
    if (this.currentStateName === name) return
    this.currentState?.exit()
    this.currentState = this.states.get(name) ?? null
    this.currentStateName = name
    this.currentState?.enter()
  }

  update(delta: number): void {
    this.currentState?.update(delta)
  }

  getCurrentState(): string {
    return this.currentStateName
  }
}
```

```ts
private position: BABYLON.Vector3 = new BABYLON.Vector3(0, 0, 10)
private speed: number = 5  // 相同数值

update(delta: number): void {
  const direction = BABYLON.Vector3.Zero()
  if (this.input.isActionActive('moveForward')) direction.z -= 1
  if (this.input.isActionActive('moveBackward')) direction.z += 1
  if (this.input.isActionActive('moveLeft')) direction.x -= 1
  if (this.input.isActionActive('moveRight')) direction.x += 1

  if (direction.length() > 0) {
    direction.normalize()
    this.position.addInPlace(direction.scaleInPlace(this.speed * delta))
    // 边界限制：相同数值
    this.position.x = Math.max(-14, Math.min(14, this.position.x))
    this.position.z = Math.max(-14, Math.min(14, this.position.z))
    this.targetRotation = Math.atan2(direction.x, direction.z)
  }

  this.mesh.position = this.position.clone()
  // 平滑旋转（相同逻辑）
  const currentRotation = this.mesh.rotation.y
  const rotationDiff = this.targetRotation - currentRotation
  const normalizedDiff = Math.atan2(Math.sin(rotationDiff), Math.cos(rotationDiff))
  this.mesh.rotation.y += normalizedDiff * 10 * delta
}
```

```ts
export function createWeaponMesh(type: PropType, scene: BABYLON.Scene): BABYLON.TransformNode {
  switch (type) {
    case 'mace': return createMaceMesh(scene)
    case 'bat': return createBatMesh(scene)
    case 'frying_pan': return createFryingPanMesh(scene)
    case 'ruler': return createRulerMesh(scene)
    default: return new BABYLON.TransformNode('weapon', scene)
  }
}

function createMaceMesh(scene: BABYLON.Scene): BABYLON.TransformNode {
  const group = new BABYLON.TransformNode('mace', scene)
  
  // 手柄：CylinderGeometry(0.04, 0.04, 0.8, 8)
  const handle = BABYLON.MeshBuilder.CreateCylinder('handle', {
    diameterTop: 0.08, diameterBottom: 0.08, height: 0.8, tessellation: 8
  }, scene)
  handle.parent = group
  
  // 头部：SphereGeometry(0.15, 8, 8)
  const head = BABYLON.MeshBuilder.CreateSphere('head', { diameter: 0.3, segments: 8 }, scene)
  head.position.y = 0.45
  head.parent = group
  
  // 8 根刺：ConeGeometry(0.04, 0.12, 4) at radius 0.15
  for (let i = 0; i < 8; i++) {
    const angle = (i / 8) * Math.PI * 2
    const spike = BABYLON.MeshBuilder.CreateCylinder('spike', {
      diameterTop: 0, diameterBottom: 0.08, height: 0.12, tessellation: 4
    }, scene)
    spike.position.set(Math.cos(angle) * 0.15, 0.45, Math.sin(angle) * 0.15)
    spike.rotation.z = Math.PI / 2
    spike.rotation.y = angle
    spike.parent = group
  }
  
  return group
}
```

- Enemy.ts 有 979 行内联状态逻辑 — StateMachine 提取将提高可维护性
- 所有游戏常量在源文件中硬编码 — 迁移期间必须保持精确值
- Part B HUD 组件是引擎无关的（纯 Vue）— 无需更改
