# 重构优化设计规划: "是男人就踹他一百下"

## 当前状态

- **技术栈**: Vue 3 + Three.js + TypeScript + Vite
- **代码量**: ~3,043 行, 6 个核心游戏文件
- **可运行**: 是, 但存在架构债务

### 发现的关键问题

| 类别 | 问题 | 严重度 |
|------|------|--------|

### 5.3 武器装备系统 — Player.ts 扩展

```typescript
// 新增属性
private equippedWeapon: WeaponConfig | null = null;
private weaponMesh: THREE.Group | null = null;
private isSwinging: boolean = false;
private swingTime: number = 0;
private swingCooldown: number = 0;

// 装备武器
equipWeapon(weapon: WeaponConfig): void {
  this.equippedWeapon = weapon;
  this.weaponMesh = weapon.meshFactory();
  // 挂载到右手
  this.armRight.add(this.weaponMesh);
  this.weaponMesh.position.set(0, -0.35, 0); // 手下方
}

// 卸下武器
unequipWeapon(): void {
  if (this.weaponMesh) {
    this.armRight.remove(this.weaponMesh);
    this.weaponMesh = null;
  }
  this.equippedWeapon = null;
}

// 挥砍攻击 (鼠标左键触发)
swingWeapon(): void {
  if (!this.equippedWeapon || this.swingCooldown > 0 || this.isSwinging) return;
  this.isSwinging = true;
  this.swingTime = this.equippedWeapon.swingDuration;
  this.swingCooldown = this.equippedWeapon.swingDuration + 0.2;
}

// update() 中的武器挥砍动画
private updateWeaponSwing(delta: number): void {
  if (this.isSwinging) {
    this.swingTime -= delta;
    const progress = 1 - (this.swingTime / this.equippedWeapon!.swingDuration);
    // 右臂从举起到挥下的弧线
    this.armRight.rotation.x = -Math.PI * 0.8 * Math.sin(progress * Math.PI);
    // 武具自身旋转增强打击感
    if (this.weaponMesh) {
      this.weaponMesh.rotation.z = Math.PI * 0.3 * Math.sin(progress * Math.PI);
    }
    if (this.swingTime <= 0) {
      this.isSwinging = false;
      this.armRight.rotation.x = 0;
      if (this.weaponMesh) this.weaponMesh.rotation.z = 0;
    }
  }
}
```

### 5.10 神人技能: 开会 — Enemy.ts + GameLoop.ts + Game.vue

**触发机制**: 神人在 `normal` 状态下有概率发动"开会"技能, 与回头检测共用调度系统

```typescript
// Enemy.ts 新增
private isMeeting: boolean = false;
private meetingTimer: number = 0;
private meetingDuration: number = 6; // 开会持续6秒
private nextMeetingChance: number = 20; // 每20秒检查一次是否开会
private meetingCooldown: number = 0;

// 开会调度 (在 updateNormal 中)
private updateNormal(delta: number) {
  // ... 现有回头逻辑 ...
  
  // 开会技能调度
  this.meetingCooldown -= delta;
  if (this.meetingCooldown <= 0) {
    this.meetingCooldown = this.nextMeetingChance;
    // 关卡越高, 开会概率越大
    const chance = 0.15 + (this.difficulty - 1) * 0.1; // 15%~45%
    if (Math.random() < chance) {
      this.startMeeting();
    }
  }
}

private startMeeting(): void {
  this.isMeeting = true;
  this.meetingTimer = this.meetingDuration;
  this.state = 'meeting'; // 新增状态
  this.events.emit('enemy:meeting', { duration: this.meetingDuration });
}
```

```typescript
  this.body.rotation.z = Math.sin(this.meetingTimer * 3) * 0.03;
  
  // 手臂做"讲话"手势
  this.meetingArmWave += delta * 5;
  // 右手抬起做指点状
  this.armRight.rotation.x = -Math.PI / 3 + Math.sin(this.meetingArmWave) * 0.2;
  this.armRight.rotation.z = Math.sin(this.meetingArmWave * 0.7) * 0.15;
  
  if (this.meetingTimer <= 0) {
    this.endMeeting();
  }
}

private endMeeting(): void {
  this.isMeeting = false;
  this.state = 'normal';
  this.meetingShield.visible = false;
  this.meetingTextSprite.visible = false;
  this.body.rotation.x = 0;
  this.body.rotation.z = 0;
  this.armRight.rotation.x = 0;
  this.armRight.rotation.z = 0;
  this.scheduleNextLookBack();
}
```

```typescript
// 攻击命中检测中增加开会判断
private checkAttackHit(): void {
  if (!player.isAttacking()) return;
  if (enemy.isLookingBack()) return; // 被发现不能攻击
  
  const dist = player.getPosition().distanceTo(enemy.getPosition());
  const range = player.hasEquippedWeapon() 
    ? player.getEquippedWeapon()!.swingRange 
    : 2.0; // 踢击范围
  
  if (dist > range) return;
  
  // 开会期间: 攻击无效, 但武器照样消耗
  if (enemy.isInMeeting()) {
    if (player.hasEquippedWeapon()) {
      // 武器打在护盾上, 弹开效果 + 消耗
      events.emit('player:attack-blocked', { reason: 'meeting' });
      player.unequipWeapon(); // 武器消耗
    }
    // 踢击也无效, 但不消耗任何东西(踢击无消耗)
    return;
  }
  
  // 正常命中逻辑...
}
```
